    public enum MeasurementType
{
    WeightAndReps = 1,    // Most weightlifting exercises

    RepsOnly = 4,         // Bodyweight exercises
    Calories = 5,         // Cardio machines
    Speed = 6,            // Sprint times
    RPE = 7               // Rate of perceived exertion
}

public enum SetQuality
{
    Perfect = 1,    // Smooth, controlled, explosive
    Good = 2,       // Slight form breakdown
    Grindy = 3,     // Tough, form breaking down
    Failed = 4,     // Couldn't complete
    Skipped = 5
}

public enum FormQuality
{
    Excellent = 1,
    Good = 2,
    Fair = 3,
    Poor = 4
}
